حل اختبار algo

التمرين الأول: أعداد مرسين

1. دالة estPremier

دالة تشوف إذا عدد معين N هو عدد أولي ولا لالا. 

Fonction estPremier(N: entier): booleen
    // دالة ترجع صح (Vrai) إذا N أولي، وغالط (Faux) إذا ماشي أولي
    Variables
        i: entier // هذا عداد نستعملوه باش نشوفو القواسم
    Début
        Si N <= 1 Alors // إذا كان N صغير على 1 ولا يساويه، باينة بلي ماشي أولي
            Retourner Faux
        FinSi
        Si N == 2 Alors // العدد 2 هو العدد الزاوج الوحيد لي أولي
            Retourner Vrai
        FinSi
        Si N % 2 == 0 Alors // إذا كان N زاوجي وكبير على 2، صافي ماشي أولي
            Retourner Faux
        FinSi
        // دوكا نشوفو القواسم الفردية برك، من 3 حتى للجذر التربيعي تاع N
        i = 3
        TantQue i * i <= N Faire 
            Si N % i == 0 Alors // إذا N يتقسم على i، صافي N ماشي أولي
                Retourner Faux
            FinSi
            i = i + 2 // نروحو للعدد الفردي لي موراه
        FinTantQue
        Retourner Vrai // إذا كملنا الحلقة وما لقينا حتى قاسم، صافي N أولي
    Fin

2. دالة estMersenne

دالة تشوف إذا عدد N هو عدد مرسين. عدد مرسين يتكتب على شكل 2p - 1، وين p لازم يكون عدد أولي. 

Fonction estMersenne(N: entier): booleen
    // دالة ترجع صح (Vrai) إذا N عدد مرسين، وغالط (Faux) إذا ماشي عدد مرسين
    Variables
        p: entier // هذا هو الأس لي نحوسو عليه (2^p - 1)
        val_p_puissance_2: entier // هذي باش نخزنو فيها N+1 لي نورمالمون تكون 2^p
        temp: entier // متغير مؤقت للحساب
    Début
        Si N <= 0 Alors // أعداد مرسين لازم يكونوا موجبين قطعا
            Retourner Faux
        FinSi

        val_p_puissance_2 = N + 1 // إذا N = 2^p - 1، صافي N+1 = 2^p

        // المرحلة الأولى: لازم نتأكدو بلي N+1 هو قوة للعدد 2
        // وفي نفس الوقت نحسبو قيمة p
        p = 0
        temp = val_p_puissance_2

        Si temp <= 0 Alors
             Retourner Faux
        FinSi

        TantQue temp % 2 == 0 ET temp > 1 Faire // ماحد temp زاوجي وكبير على 1
            temp = temp / 2
            p = p + 1
        FinTantQue

        Si temp == 1 Alors
            // المرحلة الثانية: نتأكدو بلي p لي لقيناه هو عدد أولي
            Retourner estPremier(p)
        Sinon // إذا temp ماشي 1، صافي N+1 ماكانش قوة للعدد 2
            Retourner Faux
        FinSi
    Fin

3. تعريف النوع Tab

تعريف نوع جديد "tab" يمثل جدول تاع أعداد صحيحة فيه الماكسيموم 100 عدد. 

Type Tab = Enregistrement
    elements: Tableau[1..100] d'entiers // الجدول لي فيه الأعداد
    taille_actuelle: entier // شحال من عدد كاين صح فالجدول
FinEnregistrement

4. البرنامج الرئيسي (Mersenne)

برنامج رئيسي يقرا k عدد صحيح من عند المستعمل، يحطهم في جدول A، ومن بعد يحسب ويعرض شحال كاين من عدد مرسين بيناتهم. 

Programme Principal_Mersenne
    Variables
        A: Tab // الجدول تاعنا لي راح نعمروه
        k: entier // عدد الأعداد لي راح يدخلهم المستعمل
        i: entier // عداد للحلقات
        compteur_mersenne: entier // نحسبو بيه شحال كاين من عدد مرسين
    Début
        Écrire "دخل شحال من عدد راح تقرا (k)، الماكسيموم 100: "
        Lire k

        Si k < 0 Ou k > 100 Alors // نتأكدو بلي k معقول
            Écrire "العدد k ماشي صحيح."
            Retourner // نخرجو مؤقتا
        FinSi

        A.taille_actuelle = k // نحددو الحجم الفعلي للجدول
        Écrire "دخل الأعداد الـ ", k, " تاعك واحد بواحد:"
        Pour i de 1 à k Faire
            Lire A.elements[i] // نقراو كل عدد ونحطوه فالبلاصة تاعو فالجدول
        FinPour

        compteur_mersenne = 0 // نبدو الحساب من الصفر
        Pour i de 1 à A.taille_actuelle Faire // ندورو على كامل الأعداد لي فالجدول
            Si estMersenne(A.elements[i]) Alors // إذا العدد الحالي هو عدد مرسين
                compteur_mersenne = compteur_mersenne + 1 // نزيدو واحد للعداد
            FinSi
        FinPour

        Écrire "عدد أعداد مرسين لي كاينين فالجدول هو: ", compteur_mersenne
    Fin

التمرين الثاني: المجال

1. تعريف النوع Intervalle

تعريف نوع "انتيرفال" لي يمثل مجال تاع أعداد صحيحة، عندو حد أدنى وحد أقصى. 

Type Intervalle = Enregistrement
    min: entier // الحد الأدنى للمجال
    max: entier // الحد الأقصى للمجال
FinEnregistrement

2. دالة creerIntervalle و بروسيدور afficherIntervalle

دالة تقرا من عند المستعمل حد أدنى وحد أقصى وترجع مجال جديد، وبروسيدور تافيشي مجال معين. 

Fonction creerIntervalle(): Intervalle
    // دالة تقرا من عند المستعمل حد أدنى وحد أقصى وترجع مجال (انتيرفال) جديد
    Variables
        nouvel_I: Intervalle // المجال الجديد لي راح نكرييوه
        b_min, b_max: entier // باش نقراو فيهم الحدود
    Début
        Écrire "دخل الحد الأدنى للمجال: "
        Lire b_min
        Écrire "دخل الحد الأقصى للمجال: "
        Lire b_max

        Si b_min <= b_max Alors
            nouvel_I.min = b_min
            nouvel_I.max = b_max
        Sinon // إذا دخل min كبير على max
            Écrire "تنبيه: راك دخلت حد أدنى أكبر من الحد الأقصى. المجال راح يكون فارغ حسب التعريف."
            nouvel_I.min = b_min
            nouvel_I.max = b_max
        FinSi
        Retourner nouvel_I
    Fin

Procédure afficherIntervalle(I: Intervalle)
    // بروسيدور تافيشي مجال معين على الشكل [min..max]
    Début
        Écrire "[", I.min, "..", I.max, "]"
    Fin

3. دالة intervalVide

دالة ترجع صح (Vrai) إذا كان المجال I فارغ (معناه min > max)، وترجع غالط (Faux) إذا ماشي فارغ. 

Fonction intervalVide(I: Intervalle): booleen
    // ترجع صح (Vrai) إذا المجال فارغ، وغالط (Faux) إذا لالا
    Début
        Si I.min > I.max Alors // الشرط تاع مجال فارغ كيما طلبوه
            Retourner Vrai // هذا يمثل 1
        Sinon
            Retourner Faux // هذا يمثل 0
        FinSi
    Fin

4. دالة egaux

دالة تشوف إذا زوج مجالات I1 و I2 متساويين. 

Fonction egaux(I1: Intervalle, I2: Intervalle): booleen
    Début
        Si I1.min == I2.min ET I1.max == I2.max Alors
            Retourner Vrai
        Sinon
            Retourner Faux
        FinSi
    Fin

5. مجموعة E تاع المجالات

(أ) تعريف النوع EnsembleIntervalles

تعريف نوع يمثل مجموعة تاع مجالات (جدول تاع مجالات). 

Type EnsembleIntervalles = Enregistrement
    intervalles: Tableau[1..100] d'Intervalle // جدول المجالات
    nb_elements: entier // شحال كاين من مجال صح فالمجموعة
FinEnregistrement

// Variables
//     E: EnsembleIntervalles

(ب) بروسيدورات LireEnsemble و AfficherEnsemble

بروسيدور تعمر مجموعة Ens بـ N مجال، وبروسيدور تافيشي مجموعة Ens. 

Procédure LireEnsemble(N_a_lire: entier, resultat Ens: EnsembleIntervalles)
    Variables
        i: entier
    Début
        Si N_a_lire < 0 Ou N_a_lire > 100 Alors
            Écrire "عدد المجالات لي مديتو ماشي صحيح."
            Ens.nb_elements = 0
            Retourner
        FinSi
        Ens.nb_elements = N_a_lire
        Pour i de 1 à Ens.nb_elements Faire
            Écrire "دخل معلومات المجال رقم ", i, " :"
            Ens.intervalles[i] = creerIntervalle()
        FinPour
    Fin

Procédure AfficherEnsemble(Ens: EnsembleIntervalles)
    Variables
        i: entier
    Début
        Si Ens.nb_elements == 0 Alors
            Écrire "المجموعة راهي فارغة."
            Retourner
        FinSi
        Écrire "المجالات لي كاينين فالمجموعة (" , Ens.nb_elements, " مجالات) :"
        Pour i de 1 à Ens.nb_elements Faire
            afficherIntervalle(Ens.intervalles[i])
            Si i < Ens.nb_elements Alors
                Écrire ", "
            FinSi
        FinPour
        Écrire ""
    Fin

(ج) دالة RechSeq

بحث تسلسلي: دالة تشوف إذا مجال معين I كاين فالمجموعة Ens. 

Fonction RechSeq(I_cherche: Intervalle, Ens: EnsembleIntervalles): booleen
    Variables
        i: entier
        trouve: booleen
    Début
        trouve = Faux
        i = 1
        TantQue i <= Ens.nb_elements ET NON trouve Faire
            Si egaux(Ens.intervalles[i], I_cherche) Alors
                trouve = Vrai
            FinSi
            i = i + 1
        FinTantQue
        Retourner trouve
    Fin

(د) دالة RechDich

بحث ثنائي: دالة تشوف إذا مجال I كاين فالمجموعة Ens (لي تكون مرتبة على حساب الحد الأدنى). 

Fonction RechDich(I_cherche: Intervalle, Ens: EnsembleIntervalles): booleen
    Variables
        inf, sup, milieu: entier
    Début
        inf = 1
        sup = Ens.nb_elements
        TantQue inf <= sup Faire
            milieu = partie_entiere((inf + sup) / 2)
            Si I_cherche.min < Ens.intervalles[milieu].min Alors
                sup = milieu - 1
            Sinon Si I_cherche.min > Ens.intervalles[milieu].min Alors
                inf = milieu + 1
            Sinon
                If egaux(I_cherche, Ens.intervalles[milieu]) Then
                    Retourner Vrai
                Else
                    If I_cherche.max < Ens.intervalles[milieu].max Then
                        sup = milieu - 1
                    Else
                        inf = milieu + 1
                    EndIf
                EndIf
            FinSi
        FinTantQue
        Retourner Faux
    Fin

(هـ) البرنامج الرئيسي (Intervalle) مع الفرز

برنامج رئيسي يبني مجموعة E، يرتبها، ثم يبحث فيها عن مجال I. 

// بروسيدور للفرز بالاختيار (مثال على الفرز بالتباديل)
Procédure TriParSelection_Ensemble(param_ Ens: EnsembleIntervalles)
    Variables
        i, j, idx_min_courant: entier
        temp_intervalle: Intervalle
    Début
        Pour i de 1 à Ens.nb_elements - 1 Faire
            idx_min_courant = i
            Pour j de i + 1 à Ens.nb_elements Faire
                Si Ens.intervalles[j].min < Ens.intervalles[idx_min_courant].min Alors
                    idx_min_courant = j
                FinSi
            FinPour
            Si idx_min_courant != i Alors
                temp_intervalle = Ens.intervalles[i]
                Ens.intervalles[i] = Ens.intervalles[idx_min_courant]
                Ens.intervalles[idx_min_courant] = temp_intervalle
            FinSi
        FinPour
    Fin

Programme Principal_Intervalle
    Variables
        E: EnsembleIntervalles
        N_inter: entier
        I_a_chercher: Intervalle
        est_trouve: booleen
    Début
        Écrire "دخل شحال من مجال (N) راح يكون فالمجموعة E (الماكسيموم 100): "
        Lire N_inter

        LireEnsemble(N_inter, E)

        Écrire "المجموعة E قبل ما نرتبوها:"
        AfficherEnsemble(E)

        TriParSelection_Ensemble(E)

        Écrire "المجموعة E بعد ما رتبناها:"
        AfficherEnsemble(E)

        Écrire "دوكا، دخل معلومات المجال I لي راك تحوس عليه:"
        I_a_chercher = creerIntervalle()

        est_trouve = RechDich(I_a_chercher, E)

        Si est_trouve Alors
            afficherIntervalle(I_a_chercher)
            Écrire " لقيناه فالمجموعة E."
        Sinon
            afficherIntervalle(I_a_chercher)
            Écrire " ما لقيناهش فالمجموعة E."
        FinSi
    Fin
